Новые вопросы из кода:

1. Можно добавить ignore_above для keyword полей:
yaml
filterable_keyword: &filterable_keyword
  type: keyword
  normalizer: lowercase_normalizer
  ignore_above: 256  # Игнорировать слишком длинные значения
2. Можно добавить doc_values: true для агрегаций:
yaml
filterable_keyword: &filterable_keyword
  type: keyword
  normalizer: lowercase_normalizer
  doc_values: true  # Для быстрых агрегаций и сортировки
3. Для rating можно добавить метаданные:
yaml
rating:
  type: half_float
  meta:
    description: "Рейтинг товара от 1 до 5"
    min: 1.0
    max: 5.0

python
# Найти товары с вариантом цвета Black в наличии
NestedQuery(
    path="variants",
    query=Bool(
        _must=[
            Term(field="variants.color", value="black"),
            Range(field="variants.stock", gt=0)
        ]
    )
)
